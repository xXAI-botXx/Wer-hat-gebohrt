[Doku]    def process_status(self):
        pid = self.process.pid
        process_status = None
        try:
            process_status = psutil.Process(pid).status()
        except psutil.NoSuchProcess as no_proc_exc:
            pass
            #print(no_proc_exc)

        return process_status
